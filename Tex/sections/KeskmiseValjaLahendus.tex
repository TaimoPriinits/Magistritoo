a^{ \dagger}a^{ \dagger} a a \rightarrow \left\langle  a^{ \dagger} a^{ \dagger} \right\rangle a a + a^{ \dagger} a^{ \dagger} \left\langle a a \right\rangle - \left\langle a^{ \dagger} a^{ \dagger} \right\rangle \left\langle a a \right\rangle
\end{equation}
